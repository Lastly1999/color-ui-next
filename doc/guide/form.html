# 📄 Form 表单

TIP

此组件一般用于表单场景，可以配置Input输入框，Select弹出框，进行表单验证等。

# 😐 普通表单

<template>
  <cu-form ref="cuForm" :modal="form">
    <cu-form-item label="账号" prop="userName">
      <cu-input v-model="form.userName" placeholder="请输入账号"/>
    </cu-form-item>
    <cu-form-item label="密码" prop="passWord">
      <cu-input v-model="form.passWord" placeholder="请输入密码"/>
    </cu-form-item>
    <cu-form-item label="邮箱" prop="email">
      <cu-input v-model="form.email" placeholder="请输入邮箱"/>
    </cu-form-item>
    <cu-button type="blue" @click="submit" style="width: 80%">提交</cu-button>
  </cu-form>
</template>

export default {
    data() {
        return {
            form: {
                userName: "",
                passWord: "",
                email: ''
            },
        }
    },
    methods: {
        submit() {
            // 提交后的操作
        }
    }
}

❗ 此表单demo只演示默认表单的使用方法，需要表单验证请看下方带验证的表单。

# 😄 验证表单

TIP

和普通表单的区别在于，带上内置表单验证，需要传递字段modal以及rules验证规则，规则实现的底层基于async-validator这个库，来进行异步验证

<template>
  <cu-form ref="cuForm" :rules="rules" :modal="form">
    <cu-form-item label="账号" prop="userName">
      <cu-input v-model="form.userName" placeholder="请输入账号"/>
    </cu-form-item>
    <cu-form-item label="密码" prop="passWord">
      <cu-input v-model="form.passWord" placeholder="请输入密码"/>
    </cu-form-item>
    <cu-form-item label="邮箱" prop="email">
      <cu-input v-model="form.email" placeholder="请输入邮箱"/>
    </cu-form-item>
    <cu-button type="blue" @click="submit" style="width: 80%">提交</cu-button>
  </cu-form>
</template>

export default {
    data() {
        return {
            // 字段模型
            form: {
                userName: "",
                passWord: "",
                email: ''
            },
            // 验证规则
            rules: {
                userName: [
                    {
                        type: 'string',
                        required: true,
                        message: '账号不能为空'
                    }
                ],
                passWord: [
                    {
                        type: 'string',
                        required: true,
                        message: '密码不能为空'
                    }
                ],
                email: [
                    {
                        type: 'string',
                        required: true,
                        message: '邮箱不能为空'
                    }
                ]
            }
        }
    },
    methods: {
        submit() {
            this.$refs.cuForm.validate((result) => {
               // 提交后的操作
               // result true为验证成功，false为验证失败 这是个Promise 
            })
        }
    }
}

❗❗❗ 注意：如果使用表单验证 cu-form组件的rules 以及 cu-form-item prop 和提交前的访问cu-form实例内的validate方法缺一不可，否则无法正常完成验证

# 😆 带图标的表单单元

TIP

注意：目前只有内置的组件支持扩展，如按钮，图标，头像

<template>
  <cu-form ref="cuForm" :modal="form">
    <cu-form-item label="账号" prop="userName">
      <cu-input v-model="form.userName" placeholder="请输入账号"/>
      // 直接放置自动到input尾部
      <cu-icon name="locationfill" color="orange"></cu-icon>
    </cu-form-item>
    <cu-form-item label="密码" prop="passWord">
      <cu-input v-model="form.passWord" placeholder="请输入密码"/>
      // 直接放置自动到input尾部
      <cu-button type="green" shadow>按钮</cu-button>
    </cu-form-item>
    <cu-form-item label="邮箱" prop="email">
      <cu-input v-model="form.email" placeholder="请输入邮箱"/>
      // 直接放置自动到input尾部
      <cu-button type="red" shadow>按钮</cu-button>
    </cu-form-item>
    <cu-button type="blue" @click="submit" style="width: 80%">提交</cu-button>
  </cu-form>
</template>

export default {
    data() {
        return {
            form: {
                userName: "",
                passWord: "",
                email: ''
            },
        }
    },
    methods: {
        submit() {
            // 提交后的操作
        }
    }
}
