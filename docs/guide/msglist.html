# 📔 MsgList 消息列表

TIP

该组件一般用于消息列表的使用，基于dataCom组件规范开发


<template>
    <cu-msg-list :data="data" @click="colClick"></cu-msg-list>
</template>

export default {
      data() {
            return {
                  data: [
                        {
                              url: "https://ossweb-img.qq.com/images/lol/web201310/skin/big10001.jpg",
                              title: "我是一个标题",
                              content: "测试内容1",
                              contentColor: '',
                              titleColor: '',
                              rightCt: '18:00',
                              rightCtColor: '',
                              rightBtmCt: '10',
                              rightBtmIcon: '',
                              rightBtmColor: '',
                              titleBtmIcon: '',
                              titleBtmColor: ''
                        },
                        {
                              url: "https://ossweb-img.qq.com/images/lol/web201310/skin/big10001.jpg",
                              title: "我是二个标题",
                              content: "测试内容2",
                              contentColor: '',
                              titleColor: '',
                              rightCt: '18:00',
                              rightCtColor: '',
                              rightBtmCt: '10',
                              rightBtmIcon: '',
                              rightBtmColor: '',
                              titleBtmIcon: '',
                              titleBtmColor: ''
                        },
                        {
                              url: "https://ossweb-img.qq.com/images/lol/web201310/skin/big10001.jpg",
                              title: "我是三个标题",
                              content: "测试内容3",
                              contentColor: '',
                              titleColor: '',
                              rightCt: '18:00',
                              rightCtColor: '',
                              rightBtmCt: '10',
                              rightBtmIcon: '',
                              rightBtmColor: '',
                              titleBtmIcon: '',
                              titleBtmColor: ''
                        }
                  ]
            }
      },
      methods:{
            colClick(item){
                  // 返回类型defaultProps 见下方接口说明
                  console.log(item)
            }
      }
}

interface DefaultProps {
    url: string  // 左侧头像地址
    title: string // 头像右侧上方标题内容
    content: string // 头像右侧下方内容
    contentColor: string // 头像右侧下方内容文本颜色
    titleColor: string // 头像右侧上方标题颜色
    rightCt: string // 消息单元右侧时间
    rightCtColor: string // 消息单元右侧文字内容颜色
    rightBtmCt: string // 消息单元右侧底部内容
    rightBtmIcon: string // 消息单元右侧底部图标(图标name为icon组件name)
    rightBtmColor: string // 消息单元右侧内容文本图标(图标name为icon组件name)
    titleBtmIcon: string // 消息单元右侧标题内容文本颜色(图标通用)
    titleBtmColor: string // 消息单元右侧标题内容文本颜色(图标通用)
}
